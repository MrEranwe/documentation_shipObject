class Rudder{
    constructor(position,Area,rho){
        this.x = position[0];
        this.y = position[1];
        this.z = position[2];
        this.Area = Area;
        if(typeof rho == 'undefined'){
            this.rho = 1025;
        }else{this.rho = rho;}
    }
    CalForce(attackAngle,speed){
        var F = attackAngle*Math.PI*this.rho*Math.pow(speed,2)*this.Area
        return [0,F,0];
    }

}
